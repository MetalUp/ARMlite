

//Input string
start:
MOV R0, #string1
STR R0, .ReadString
MOV R3, #0  //Count of characters

//Push string onto stack
loop1: LDRB R1, [R0]
CMP R1, #0
BEQ next
PUSH R1
ADD R3, R3, #1
ADD R0, R0, #1
B loop1

next:
MOV R0, #string1 //Reset index
loop2: CMP R3, #0
BEQ done
POP R1
STRB R1, [R0]
ADD R0, R0, #1
SUB R3, R3, #1
B loop2

done:
MOV R0, #string1
STR R0, .WriteString
B start

string1: .BLOCK 128
